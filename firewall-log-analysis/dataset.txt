# SCENARIO 1 – PORT SCANNING DETECTION

| makeresults count=200
| streamstats count as n

| eval src_ip=case(
n<120,"192.168.100.10",
true(),"192.168.100.50"
)

| eval dest_ip="10.10.10.5"

| eval dest_port=case(
n<20,21,
n<40,22,
n<60,23,
n<80,25,
n<100,53,
n<120,80,
true(),443
)

| eval action=case(
n<120,"DENY",
true(),"ALLOW"
)

| table src_ip dest_ip dest_port action
